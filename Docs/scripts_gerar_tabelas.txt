Pessoa
CREATE TABLE Pessoa
(
	IdPessoa int IDENTITY(1,1) PRIMARY KEY,
	Nome VARCHAR(100) NOT NULL,
	CPF varchar(11) NOT NULL UNIQUE,
	Data_Nascimento DATETIME NOT NULL,
	Telefone VARCHAR(14) NOT NULL,
	Email VARCHAR(100) NOT NULL,
	Endereco VARCHAR(100) NOT NULL,
	Genero VARCHAR(1) NOT NULL,
	RefreshToken VARCHAR(MAX),
	RefreshTokenExpiryTime DATETIME2
)
Permissões
CREATE TABLE Permissoes
(
	IdPermissao INT IDENTITY(1,1) PRIMARY KEY,
	descricao VARCHAR(30) NOT NULL
)
PessoaPermissao
CREATE TABLE PessoaPermissao
(
	IdPermissao INT NOT NULL,
	IdPessoa int NOT NULL

	CONSTRAINT FK_IdPessoa
	FOREIGN KEY(IdPessoa)
	REFERENCES Pessoa(IdPessoa),

	CONSTRAINT FK_IdPermissao
	FOREIGN KEY(IdPermissao)
	REFERENCES Permissoes(IdPermissao)
)
CLIENTE
	CREATE TABLE Clientes
	(
		IdCliente int PRIMARY KEY,
		NumeroConta varchar(15) UNIQUE NOT NULL,
		Saldo DECIMAL DEFAULT 0 NOT NULL,
		ATIVO BIT DEFAULT 1 NOT NULL,
		Senha VARCHAR(50) NOT NULL,
		DataCriacao DATETIME DEFAULT GETDATE() NOT NULL

		CONSTRAINT FK_IdCliente_Cliente
		FOREIGN KEY(IdCliente)
		REFERENCES Pessoa(IdPessoa)
	)
Funcionários
CREATE TABLE Funcionarios
(
	IdFuncionario int PRIMARY KEY,
	Cargo VARCHAR(20) NOT NULL,
	Tipo BIT NOT NULL,
	DocumentoAdimissao VARCHAR(14),
	DataAdimissao DATETIME DEFAULT GETDATE(),
	SALARIO DECIMAL NOT NULL,
	
	CONSTRAINT FK_IdFuncionario
	FOREIGN KEY(IdFuncionario)
	REFERENCES Pessoa(IdPessoa)
)
Cofrinho
CREATE TABLE Cofrinho
(
	IdCofre UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
	IdPessoa int NOT NULL,
	Saldo DECIMAL NOT NULL DEFAULT 0,
	DataCriacao DATE NOT NULL DEFAULT GETDATE(),
	Nome Varchar(100) NOT NULL,
	Descricao VARCHAR(100),

	CONSTRAINT FK_IdPessoa_Cofrinho
	FOREIGN KEY(IdPessoa)
	REFERENCES Pessoa(IdPessoa)
)
